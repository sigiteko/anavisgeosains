import numpy as np
import matplotlib.pyplot as plt
from matplotlib.colors import ListedColormap, Normalize
from matplotlib.cm import ScalarMappable
from matplotlib.lines import Line2D
from mpl_toolkits.basemap import Basemap
from obspy.imaging.beachball import beach
import pandas as pd

m = Basemap(projection='merc',resolution = 'c', area_thresh = 1000.0,llcrnrlon=min(df['longitude']), llcrnrlat=min(df['latitude'])-10, #min longitude (llcrnrlon) and latitude (llcrnrlat)
            urcrnrlon=max(df['longitude']), urcrnrlat=max(df['latitude'])+10, epsg=4326)

m.drawcoastlines()
m.fillcontinents()
m.drawparallels(np.arange(-90., 120., 30.),labels=[True,False,False,True])
m.drawmeridians(np.arange(0., 420., 60.),labels=[True,False,False,True])
m.drawmapboundary()

# Define custom colormap based on earthquake depth
colors = ['red', 'green', 'blue']
cmap = ListedColormap(colors)
norm = plt.Normalize(0, 60)  # Red for depth 0-60 km

ax = plt.gca()

# Loop melalui dataframe dan gambar beachball untuk setiap pasangan x dan y
for index, row in df.iterrows():
    x, y = m(row['longitude'], row['latitude'])
    focmecs = [row['mrr'], row['mtt'], row['mpp'], row['mrt'], row['mrp'], row['mtp']]
    b = beach(focmecs, xy=(x, y), width=row['magnitudes_MS']*1.3, linewidth=1, facecolor=cmap(norm(row['depth'])), alpha=0.7)
    b.set_zorder(10)
    ax.add_collection(b)


legend_elements = [Line2D([0], [0], marker='o', color='w', label='Gempabumi',
                          markerfacecolor='darkred', markersize=7)]

# Create the figure
judul='Episenter Gempabumi Berdasarkan eq_catalogue.ndk'
plt.legend(handles=legend_elements, loc='upper right')
plt.title(judul)

# Create a colorbar
sm = ScalarMappable(cmap=cmap, norm=norm)
sm.set_array([])
cbar = plt.colorbar(sm, ticks=[0, 30, 60, 300], pad=0.05, shrink=0.5)
cbar.set_ticklabels(['0 km', '60-300 km', '>300 km', '>300 km'])
cbar.set_label('Earthquake Depth')

plt.show()
